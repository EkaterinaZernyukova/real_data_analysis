Полученное статистическое значение -4.462995 меньше, чем -3.456 при 1%, что позволяет отклонить нулевую гипотезу с уровнем значимости менее 1% (низкой вероятностью статистической случайности). Временной ряд является стационарным

Для стационарного временного ряда будет использована Модель авторегрессии — скользящего среднего (ARIMA).

X = df.revenue.values
size = int(len(X) * 0.5)
train, test = X[0:size], X[size:len(X)]
history = [x for x in train]
predictions = list()
for t in range(len(test)):
    model = ARIMA(history, order=(5, 1, 0))
    model_fit = model.fit(disp=0)
    output = model_fit.forecast()
    yhat = output[0]
    predictions.append(yhat)
    obs = test[t]
    history.append(obs)
#     print('predicted=%f, expected=%f' % (yhat, obs))
error = mean_squared_error(test, predictions)
print('Test MSE: %.3f' % error)
# plot
pyplot.plot(test)
pyplot.plot(predictions, color='red')
pyplot.show()
